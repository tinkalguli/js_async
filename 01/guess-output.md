    now = Date.now();
  }
}
function main() {
  console.log("A");
  setTimeout(function exec() {
    console.log("B");

  let start = Date.now(),

    now = Date.now();
  }
}

  var current = Date.now();
  console.log("A", Date.now() - current);
  setTimeout(function exec() {

}
